/**
 * Animation Utilities
 *
 * Easing functions for smooth animations in the sphere grid
 * and other landing page components.
 */

/**
 * Ease out cubic - fast start, slow end
 */
export function easeOutCubic(t: number): number {
  return 1 - Math.pow(1 - t, 3);
}

/**
 * Ease in out sine - smooth S-curve
 */
export function easeInOutSine(t: number): number {
  return -(Math.cos(Math.PI * t) - 1) / 2;
}

/**
 * Ease in quadratic - slow start
 */
export function easeInQuad(t: number): number {
  return t * t;
}

/**
 * Ease out quadratic - slow end
 */
export function easeOutQuad(t: number): number {
  return 1 - (1 - t) * (1 - t);
}

/**
 * Ease in out cubic - symmetric S-curve
 */
export function easeInOutCubic(t: number): number {
  return t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2;
}

/**
 * Sleep helper for async animations
 */
export function sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms));
}

/**
 * Typewriter effect for text elements
 */
export async function typewrite(
  element: HTMLElement,
  text: string,
  speed: number = 50
): Promise<void> {
  element.textContent = '';
  for (const char of text) {
    element.textContent += char;
    await sleep(speed + Math.random() * 20);
  }
}

export default {
  easeOutCubic,
  easeInOutSine,
  easeInQuad,
  easeOutQuad,
  easeInOutCubic,
  sleep,
  typewrite,
};
