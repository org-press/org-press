/**
 * Bitmap Font Patterns
 *
 * Simple 5x7 bitmap font for A-Z letters plus some symbols.
 * Used by the sphere grid to display text.
 */

export type CharPattern = boolean[][];
export type CharPatterns = Record<string, CharPattern>;

/**
 * Helper to convert string pattern to boolean array
 */
function p(rows: string[]): CharPattern {
  return rows.map(row => row.split('').map(c => c === '#'));
}
