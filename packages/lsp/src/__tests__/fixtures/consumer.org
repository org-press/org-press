* Combined Usage

#+name: combined
#+begin_src typescript
import { capitalize, slugify } from "./utils/string-utils";
import { clamp } from "./utils/math-utils";
import { unique } from "./utils/array-utils";

function processData(items: string[]): string[] {
  return unique(items)
    .map(item => capitalize(item))
    .slice(0, clamp(items.length, 1, 5));
}

const result = processData(["apple", "banana", "apple", "cherry"]);
console.log(result);
#+end_src
