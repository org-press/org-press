* Classes

#+name: classes
#+begin_src typescript
class Calculator {
  private value: number = 0;

  add(n: number): this {
    this.value += n;
    return this;
  }

  subtract(n: number): this {
    this.value -= n;
    return this;
  }

  getResult(): number {
    return this.value;
  }
}

const calc = new Calculator();
calc.add(5).subtract(2);
